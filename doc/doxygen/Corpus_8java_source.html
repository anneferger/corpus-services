40  public Corpus(URL url) throws MalformedURLException, MalformedURLException, MalformedURLException, SAXException, JexmaraldaException, URISyntaxException {
41  CorpusIO cio = new CorpusIO();
42  cdc = cio.read(url);
43  for (CorpusData cd : cdc) {
44  if (cd instanceof ContentData) {
45  contentdata.add((ContentData) cd);
46  } else if (cd instanceof Recording) {
47  recording.add((Recording) cd);
48  } else if (cd instanceof AdditionalData) {
49  additionaldata.add((AdditionalData) cd);
50  } else if (cd instanceof Metadata) {
51  metadata.add((Metadata) cd);
52  } else if (cd instanceof AnnotationSpecification) {
53  annotationspecification.add((AnnotationSpecification) cd);
54  } else if (cd instanceof ConfigParameters) {
55  configparameters.add((ConfigParameters) cd);
56  } else if (cd instanceof CmdiData) {
57  cmdidata.add((CmdiData) cd);
58  }
59  }
