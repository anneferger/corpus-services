276  Vector returnValue = new Vector();

279  // go through all top level segment chains
280  for (Object segmentChain : segmentChains) {
281  Element sc = (Element) (segmentChain);
282  sc.setAttribute("speaker", sc.getParentElement().getParentElement().getAttributeValue("speaker"));
283  String tierref = sc.getParentElement().getAttributeValue("tierref");
284  String start = sc.getAttributeValue("s");

286  String xpath2 = "//segmentation[@name='" + nameOfFlatSegmentation + "' and @tierref='" + tierref + "']"

291  //this means that no corresponding top level
292  //element was found in the second segmentation
293  //which should not happen
294  throw new Exception(tierref + " " + start + " " + end);
295  }
296  // this is where the magic happens
297  Element mergedElement = merge(sc, sc2);
298 
299  // now take care of the corresponding annotations
300  int s = ((Integer) (timelineItems.get(start)));
301  int e = ((Integer) (timelineItems.get(end)));
302  String xpath5 = "//segmented-tier[@id='" + tierref + "']/annotation/ta";
303  XPath xp5 = XPath.newInstance(xpath5);
304  List annotations = xp5.selectNodes(segmentedTranscription);
305  for (Object annotation1 : annotations) {
306  Element anno = (Element) (annotation1);
307  String aStart = anno.getAttributeValue("s");
308  String aEnd = anno.getAttributeValue("e");
309  int as = ((Integer) (timelineItems.get(aStart)));
310  int ae = ((Integer) (timelineItems.get(aEnd)));
311  boolean annotationBelongsToThisElement = (as >= s && as <= e) || (ae >= s && ae <= e);
312  if (annotationBelongsToThisElement) {
313  Element annotationsElement = mergedElement.getChild("annotations");
314  if (annotationsElement == null) {
315  annotationsElement = new Element("annotations");
316  mergedElement.addContent(annotationsElement);
317  }
318  Element annotation = new Element("annotation");
319  annotation.setAttribute("start", aStart);
320  annotation.setAttribute("end", aEnd);
321  annotation.setAttribute("level", anno.getParentElement().getAttributeValue("name"));
322  annotation.setAttribute("value", anno.getText());
323  annotationsElement.addContent(annotation);
324  }

326  //System.out.println(s + "/" + e + " **** " + as + "/" + ae);
327  }

369  static Element merge(Element e1, Element e2) {
370 
371  Iterator i1 = e1.getDescendants();
372  Vector pcData1 = new Vector();
373  while (i1.hasNext()) {
374  pcData1.addElement(i1.next());
375  }
376 
377  Iterator i2 = e2.getDescendants(new TextFilter());
378  Vector pcData2 = new Vector();
379  while (i2.hasNext()) {
380  pcData2.addElement(i2.next());
381  }
382 
383  int charBoundary = 0;
384  for (int pos = 0; pos < pcData2.size() - 1; pos++) {
385  Text eventText = (Text) (pcData2.elementAt(pos));
386  Element anchor = new Element("anchor");
387  Element event = eventText.getParentElement();
388  String start = event.getAttributeValue("e");
389  anchor.setAttribute("synch", start);
390 
391  charBoundary += eventText.getText().length();
392  // jetzt durch den anderen Baum laufen und den zugehoerigen Anker
393  // an der richtigen Stelle einfuegen
394  int charCount = 0;
395  for (int pos2 = 0; pos2 < pcData1.size(); pos2++) {
396  Object o = pcData1.elementAt(pos2);
397  if (!(o instanceof Text)) {
398  continue;
399  }
400  Text segmentText = (Text) o;
401  int textLength = segmentText.getText().length();
402  if (charCount + textLength < charBoundary) {
403  charCount += textLength;
404  continue;
405  } else if (charCount + textLength == charBoundary) {
406  Element parent = segmentText.getParentElement();
407  int index = parent.indexOf(segmentText);
408  Element parentOfParent = parent.getParentElement();
409  int index2 = parentOfParent.indexOf(parent);
410  parentOfParent.addContent(index2 + 1, anchor);
411  break;
412  }
413  // charCount+textLength>charBoundary
414  String leftPart = segmentText.getText().substring(0, charBoundary - charCount);
415  String rightPart = segmentText.getText().substring(charBoundary - charCount);
416  Text leftText = new Text(leftPart);
417  Text rightText = new Text(rightPart);
418 
419  // neue Sachen muessen zweimal eingefuegt werden - einmal
420  // in den Vector, einmal in den Parent
421  // Sachen im Vector muessen den richtigen Parent bekommen
422  Element parent = segmentText.getParentElement();
423  parent.removeContent(segmentText);
424  parent.addContent(leftText);
425  parent.addContent(anchor);
426  parent.addContent(rightText);
427 
428  pcData1.remove(segmentText);
429  pcData1.add(pos2, rightText);
430  pcData1.add(pos2, anchor);
431  pcData1.add(pos2, leftText);
432  break;
433  }
434  }
435 
436  return e1;
437  }
