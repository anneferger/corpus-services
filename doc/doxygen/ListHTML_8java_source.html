78  private String createFromBasicTranscription(String btAsString, String segmAlgorithm) {
79 
80  basicTranscriptionString = btAsString;
81  basicTranscription = TypeConverter.String2BasicTranscription(basicTranscriptionString);
82  segmentationAlgorithm = segmAlgorithm;
83 
84  String result = null;
85 
86  try {
87 
88  // create an utterance list as XML basis for transformation
89  createUtteranceList();
90  if (getUtteranceList()!=null){
91  // get the XSLT stylesheet
92  String xsl = "";
93  if (segmAlgorithm.equals("HIAT")) {
94  xsl = TypeConverter.InputStream2String(
95  getClass().getResourceAsStream(STYLESHEET_PATH));
96  } else if (segmAlgorithm.equals("HIATINEL")) {
97  xsl = TypeConverter.InputStream2String(
98  getClass().getResourceAsStream(INEL_STYLESHEET_PATH));
99  } else if (segmAlgorithm.equals("GAT")) {
100  xsl = TypeConverter.InputStream2String(
101  getClass().getResourceAsStream(GAT_STYLESHEET_PATH));
102  } else if (segmAlgorithm.equals("Generic")) {
103  xsl = TypeConverter.InputStream2String(
104  getClass().getResourceAsStream(GENERIC_STYLESHEET_PATH));
105  } else {
106  xsl = TypeConverter.InputStream2String(
107  getClass().getResourceAsStream(GENERIC_STYLESHEET_PATH));
108  }
109 
110  // create XSLTransformer and set the parameters
111  XSLTransformer xt = new XSLTransformer();
112  xt.setParameter("EMAIL_ADDRESS", EMAIL_ADDRESS);
113  xt.setParameter("WEBSERVICE_NAME", SERVICE_NAME + " (" + segmAlgorithm + ")");
114  xt.setParameter("HZSK_WEBSITE", HZSK_WEBSITE);
115  xt.setParameter("TRANSCRIPTION_NAME", cd.getFilenameWithoutFileEnding());
116  //xt.setParameter("CORPUS_NAME", cd.getFilenameWithoutFileEnding());
117  if(!corpusname.equals("")){
118  xt.setParameter("CORPUS_NAME", corpusname);
119  }
120  // perform XSLT transformation
121  result = xt.transform(getUtteranceList(), xsl);
122  if (result != null) {
123 
124  // replace JS/CSS placeholders from XSLT output
125  try {
126  Pattern regex = Pattern.compile("(<hzsk\\-pi:include( xmlns:hzsk\\-pi=\"https://corpora\\.uni\\-hamburg\\.de/hzsk/xmlns/processing\\-instruction\")?>([^<]+)</hzsk\\-pi:include>)", Pattern.DOTALL);
127  Matcher m = regex.matcher(result);
128  StringBuffer sb = new StringBuffer();
129  while (m.find()) {
130  String insertion = TypeConverter.InputStream2String(getClass().getResourceAsStream(m.group(3)));
131  m.appendReplacement(sb, m.group(0).replaceFirst(Pattern.quote(m.group(1)), insertion));
132  }
133  m.appendTail(sb);
134  result = sb.toString();
135  } catch (Exception e) {
136  setHTML("Custom Exception for inserting JS/CSS into result: " + e.getLocalizedMessage() + "\n" + result);
137  stats.addException(SERVICE_NAME, e, "JS/CSS exception");
138  }
139  } else {
140  stats.addCritical(SERVICE_NAME, cd, "Visualization of file was not possible! (empty result)");
141  }
142  } else {
143  stats.addCritical(SERVICE_NAME, cd, "Visualization of file was not possible! (empty utterance list)");
144  }
145  } catch (TransformerConfigurationException ex) {
146  stats.addException(SERVICE_NAME, ex, "Transformer exception");
147  } catch (TransformerException ex) {
148  stats.addException(SERVICE_NAME, ex, "Transformer exception");
149  }
150 
151  setHTML(result);
152  //System.out.println(result);
153  return result;
154  }

163  private void createUtteranceList() {
164 
165  String list = null;
166 
167  try {
168 
169  switch (segmentationAlgorithm) {
170  case "HIAT": {
171  HIATSegmentation hS = new HIATSegmentation();
172  ListTranscription lt = hS.BasicToUtteranceList(basicTranscription);
173  final Document listXML = FileIO.readDocumentFromString(lt.toXML());
174  list = IOUtilities.documentToString(listXML);
175  break;
176  }
177  case "HIATINEL": {
178  File file = null;
179  HIATSegmentation hSi = new HIATSegmentation(INEL_FSM);
180  //hSi.utteranceFSM = INEL_FSM;
181  ListTranscription lt = hSi.BasicToUtteranceList(basicTranscription);
182  //System.out.println("ListTranscription:" + lt.toXML());
183  final Document listXML = FileIO.readDocumentFromString(lt.toXML());
184  list = IOUtilities.documentToString(listXML);
185  break;
186  }
187  case "CHAT": {
188  CHATSegmentation cS = new CHATSegmentation();
189  ListTranscription lt = cS.BasicToUtteranceList(basicTranscription);
190  final Document listXML = FileIO.readDocumentFromString(lt.toXML());
191  list = IOUtilities.documentToString(listXML);
192  break;
193  }
194  case "GAT": {
195  GATSegmentation gS = new GATSegmentation();
196  ListTranscription lt = gS.BasicToIntonationUnitList(basicTranscription);
197  final Document listXML = FileIO.readDocumentFromString(lt.toXML());
198  list = IOUtilities.documentToString(listXML);
199  break;
200  }
201  case "IPA": {
202  IPASegmentation ipaS = new IPASegmentation();
203  SegmentedTranscription st = ipaS.BasicToSegmented(basicTranscription);
204  ListTranscription lt = st.toListTranscription(new SegmentedToListInfo(st, SegmentedToListInfo.TURN_SEGMENTATION));
205  final Document listXML = FileIO.readDocumentFromString(lt.toXML());
206  list = IOUtilities.documentToString(listXML);
207  break;
208  }
209  case "Generic": {
210  GenericSegmentation genS = new GenericSegmentation();
211  SegmentedTranscription st = genS.BasicToSegmented(basicTranscription);
212  ListTranscription lt = st.toListTranscription(new SegmentedToListInfo(st, SegmentedToListInfo.TURN_SEGMENTATION));
213  final Document listXML = FileIO.readDocumentFromString(lt.toXML());
214  list = IOUtilities.documentToString(listXML);
215  break;
216  }
217  default:
218  throw new Exception("createUtteranceList - unsupported parameter segmAlgorithm='" + segmentationAlgorithm + "'");
219  }
220 
221  } catch (SAXException ex) {
222  stats.addException(SERVICE_NAME, ex, "Transformer exception");
223  } catch (FSMException ex) {
224  stats.addException(SERVICE_NAME, ex, "Transformer exception");
225  } catch (Exception ex) {
226  stats.addException(SERVICE_NAME, ex, "Exception");
227  }
228 
229  setUtteranceList(list);
230  }
