54  @Override
55  public Report function (CorpusData cd, Boolean fix) throws IOException, SAXException {
56  Document doc = null;
57  XMLData xml = (XMLData)cd;
58  doc = TypeConverter.JdomDocument2W3cDocument(xml.getJdom());
59  Report stats = new Report();
60 
61  NodeList tiers = doc.getElementsByTagName("tier");
62  ArrayList<Element> relevantTiers = new ArrayList();
63  for (int i = 0; i < tiers.getLength(); i++) {
64  Element tier = (Element)tiers.item(i);
65  String category = tier.getAttribute("category");
66  if (lsTiersToCheck.contains(category)) {
67  relevantTiers.add(tier);
68  }
69  }
70  for (int i = 0; i < relevantTiers.size(); i++) {
71  Element curTier = relevantTiers.get(i);
72  NodeList events = curTier.getElementsByTagName("event");
73  String tierId = curTier.getAttribute("id");
74 
75  for (int j = 0; j < events.getLength(); j++) {
76  Element event = (Element)events.item(j);
77  String eventStart = event.getAttribute("start");
78  String eventEnd = event.getAttribute("end");
79  String eventText = event.getTextContent();
80  Matcher m = forbSymbEx.matcher(eventText);
81  if (m.find()) {
82  forbidden = true;
83  String error = "Forbidden symbol '" + m.group() + "' was found in an event: "
84  + eventStart + "/" + eventEnd + ", tier '" + tierId + "'";
85  stats.addCritical(function, cd, error);
86  }
87  }
88  }
89 
90  if (!forbidden) {
91  stats.addCorrect(function, cd, "CorpusData file does not contain forbidden symbols");
92  }
93 
94  return stats;
95  }
