74  public Report exceptionalCheck(File f)
75  throws SAXException, IOException, ParserConfigurationException {
76  DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
77  DocumentBuilder db = dbf.newDocumentBuilder();
78  Document doc = db.parse(f);
79  NodeList reffiles = doc.getElementsByTagName("referenced-file");
80  int reffilesFound = 0;
81  int reffilesMissing = 0;
82  Report stats = new Report();
83  for (int i = 0; i < reffiles.getLength(); i++) {
84  Element reffile = (Element) reffiles.item(i);
85  String url = reffile.getAttribute("url");
86  if (url.startsWith("file:///C:") || url.startsWith("file:/C:")) {
87  stats.addCritical(EXB_REFS, cd, "Referenced-file " + url + " points to absolute local path, fix it to relative first");
88  }
89  File justFile = new File(url);
90  boolean found = false;
91  if (justFile.exists()) {
92  found = true;
93  }
94  String relfilename = url;
95  if (url.lastIndexOf("/") != -1) {
96  relfilename = url.substring(url.lastIndexOf("/"));
97  }
98  String referencePath = f.getParentFile().getCanonicalPath();
99  String absPath = referencePath + File.separator + relfilename;
100  File absFile = new File(absPath);
101  if (absFile.exists()) {
102  found = true;
103  }
104  if (!found) {
105  reffilesMissing++;
106  stats.addCritical(EXB_REFS, cd,
107  "File in referenced-file NOT found: " + url);
108  exmaError.addError(EXB_REFS, f.getName(), "", "", false, "Error: File in referenced-file NOT found: " + url);
109  } else {
110  reffilesFound++;
111  stats.addCorrect(EXB_REFS, cd, "File in referenced-file was found: " + url);
112  }
113  }
114  return stats;
115  }
