69  DocumentBuilder db = dbf.newDocumentBuilder();

73  String corpusVersion = "";
74  for (int i = 0; i < keys.getLength(); i++) {
75  Element keyElement = (Element)keys.item(i);

77  corpusPrefix = keyElement.getTextContent();

87  "PID length cannot be estimated accurately. " +

89  corpusPrefix = "muster";

95  stats.addWarning(COMA_PID_LENGTH + "-config", comaLoc + ": " +

97  "PID length cannot be estimated accurately. " +
98  "Add that key in coma.");
99  corpusVersion = "0.0";

116  }

118  stats.addCritical(COMA_PID_LENGTH, comaLoc + ": " +

121  " It must be shortened, e.g. use: " +
122  shortenedCommuniationName + ", or change " +

124  } else {
125  stats.addCorrect(COMA_PID_LENGTH, comaLoc + ": " +
126  "Following PID will be generated for this " +
127  "communication in Fedora: " + fedoraPID);

129  }
