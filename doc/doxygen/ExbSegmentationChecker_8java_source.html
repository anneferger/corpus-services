150  } else if (segmentationName.equals("CHAT_MINIMAL")) {
151  segmentation = new org.exmaralda.partitureditor.jexmaralda.segment.CHATMinimalSegmentation();
152  } else if (segmentationName.equals("DIDA")) {
153  segmentation = new org.exmaralda.partitureditor.jexmaralda.segment.DIDASegmentation();
154  } else if (segmentationName.equals("IPA")) {
155  segmentation = new org.exmaralda.partitureditor.jexmaralda.segment.IPASegmentation();
156  } else {
157  segmentation = new org.exmaralda.partitureditor.jexmaralda.segment.GenericSegmentation();
158  }
159  if (!path2ExternalFSM.equals("")) {
160  segmentation.pathToExternalFSM = path2ExternalFSM;
161  }
162  List v = segmentation.getSegmentationErrors(btd.getEXMARaLDAbt());
163  for (Object o : v) {

165  String text = fsme.getMessage();
166  stats.addCritical(EXB_SEG, cd, text);
167  exmaError.addError(EXB_SEG, filename, fsme.getTierID(), fsme.getTLI(), false, text);
168  }
