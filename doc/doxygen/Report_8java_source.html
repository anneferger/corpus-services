313  public String getSummaryLine(String statId) {
314  int good = 0;
315  int severe = 0;
316  int badish = 0;
317  int unk = 0;
318  Collection<ReportItem> stats = statistics.get(statId);
319  for (ReportItem s : stats) {
320  if (s.isSevere()) {
321  severe += 1;
322  } else if (s.isBad()) {
323  badish += 1;
324  } else if (s.isGood()) {
325  good += 1;
326  } else {
327  unk += 1;
328  }
329  }
330  int totes = good + severe + badish + unk;
331  return MessageFormat.format(" {0}: {1} %: {2} OK, {3} bad, "
332  + "{4} warnings and {5} unknown. "
333  + "= {6} items.\n", statId, 100 * good / totes,
334  good, severe, badish, unk, totes);
335  }
